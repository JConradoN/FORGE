'; } } function renderImoveis(data) { grid.innerHTML = ''; data.forEach(imovel => { const card = document.createElement('div'); card.className = 'card'; card.innerHTML = `

${imovel.endereco}

${imovel.area_m2}m² | ${imovel.quartos} quartos

R$ ${imovel.preco.toLocaleString('pt-BR')}

`; grid.appendChild(card); }); } // Filters filterButtons.forEach(btn => { btn.addEventListener('click', () => { const tipo = btn.dataset.tipo; document.querySelector('.filter-btn.active').classList.remove('active'); btn.classList.add('active'); if (tipo === 'Todos') { renderImoveis(allImoveis); } else { const filtered = allImoveis.filter(i => i.tipo === tipo); renderImoveis(filtered); } }); }); // Form Validation const form = document.getElementById('contactForm'); form.addEventListener('submit', (e) => { e.preventDefault(); let isValid = true; const nome = document.getElementById('nome'); const email = document.getElementById('email'); const mensagem = document.getElementById('mensagem'); // Reset errors document.querySelectorAll('.error-msg').forEach(el => el.style.display = 'none'); if (!nome.value.trim()) { document.getElementById('err-nome').style.display = 'block'; isValid = false; } if (!email.checkValidity()) { document.getElementById('err-email').style.display = 'block'; isValid = false; } if (!mensagem.value.trim()) { document.getElementById('err-mensagem').style.display = 'block'; isValid = false; } if (isValid) { alert('Mensagem enviada com sucesso!'); form.reset(); } }); loadImoveis();
